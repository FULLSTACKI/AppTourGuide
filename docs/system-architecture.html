  onError,

    timeout: 15000,             // Timeout sau 15 giay
  }

// Tra ve: { position, error, isTracking }

3.2 Cong thuc Haversine

De tinh khoang cach giua 2 toa do GPS (nguoi dung va dia diem), he thong su dung cong thuc Haversine - cong thuc tinh khoang cach duong tron lon tren mat cau Trai Dat voi ban kinh R = 6,371,000 met. 

// haversine.ts
const EARTH_RADIUS_METERS = 6_371_000;

export function haversineDistance(
  lat1: number, lon1: number,
  lat2: number, lon2: number
): number {
  const dLat = toRadians(lat2 - lat1);
  const dLon = toRadians(lon2 - lon1);

  const a =
    Math.sin(dLat / 2) ** 2 +
    Math.cos(toRadians(lat1)) *
    Math.cos(toRadians(lat2)) *
    Math.sin(dLon / 2) ** 2;

  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  return EARTH_RADIUS_METERS * c;
}

3.3 Ban kinh kich hoat (Trigger Radius)

AUTOPLAY_RADIUS_METERS = 15
Khi khoang cach tinh duoc < 15 met, he thong tu dong kich hoat tinh nang doc gioi thieu bang giong noi cho dia diem do. Moi dia diem chi duoc tu dong doc mot lan duy nhat trong phien su dung (theo doi bang narratedRef: Set<string>). 
// TourMapPage.tsx - Proximity Check
const AUTOPLAY_RADIUS_METERS = 15;

const checkProximitySpeech = useCallback(() => {
  if (!audioUnlocked || !position || places.length === 0) return;

  for (const place of places) {
    if (narratedRef.current.has(place.id)) continue;

    const dist = haversineDistance(
      position.latitude, position.longitude,
      place.latitude, place.longitude
    );

    if (dist < AUTOPLAY_RADIUS_METERS && place.translation?.description) {
      narratedRef.current.add(place.id);
      setSelectedPlace(place);
      speakText(`${place.translation.name}. ${place.translation.description}`);
      break; // chi trigger mot dia diem moi lan
    }
  }
}, [audioUnlocked, position, places, speakText]);

Section 05

Module Audio

5.1 Cong nghe: Web Speech API

Thay vi su dung file MP3 (ton kem luu tru, can tao truoc cho tung ngon ngu), he thong su dung window.speechSynthesis - API Text-to-Speech co san trong moi trinh duyet hien dai. Uu diem:

  • Chi phi = 0: Khong can dich vu TTS tra phi nhu Google Cloud TTS hay Amazon Polly.
  • Hoat dong offline: Nhieu thiet bi da tai san giong noi offline (dac biet iOS va Android).
  • Da ngon ngu tu dong: Chi can thay doi utterance.lang la chuyen giong doc.
// PlaceDetailSheet.tsx - Toggle Speech
const SPEECH_LANG_MAP: Record<string, string> = {
  en: 'en-US',
  vi: 'vi-VN',
  ja: 'ja-JP',
  ko: 'ko-KR',
  zh: 'zh-CN',
};

const utterance = new SpeechSynthesisUtterance(
  `${place.translation.name}. ${place.translation.description}`
);
utterance.lang = SPEECH_LANG_MAP[lang] ?? 'en-US';
utterance.rate = 0.95;
window.speechSynthesis.speak(utterance);

5.2 Bypass Autoplay Policy

Van de: Cac trinh duyet hien dai (Chrome, Safari) cam phat am thanh ma khong co tuong tac cua nguoi dung (user gesture). Dieu nay nghia la khi GPS phat hien nguoi dung gan dia diem, he thong khong the tu dong doc neu chua co click/tap truoc do.

Giai phap: Khi PWA khoi dong, hien thi overlay "Start Audio Tour" bat buoc nguoi dung nhan. Khi nhan, he thong phat mot SpeechSynthesisUtterance rong (silent) de "mo khoa" engine. Sau do, GPS co the kich hoat doc bat ky luc nao. 
// TourMapPage.tsx - Unlock Audio Engine
const handleUnlockAudio = useCallback(() => {
  // Phat utterance rong tu user gesture de thoa man browser policy
  const silent = new SpeechSynthesisUtterance('');
  window.speechSynthesis.speak(silent);
  setAudioUnlocked(true);  // Cho phep auto-narration tu day
}, []);
Section 06

Phan quyen Bearer Token

6.1 Tong quan co che xac thuc

He thong su dung Session-based Bearer Token (khong phai JWT). Token duoc tao khi dang nhap thanh cong, luu trong bang SessionTokens trong database, va duoc kiem tra moi request thong qua middleware tuy chinh.

6.2 Luong xac thuc

BuocComponentMo ta
1 POST /auth/login Frontend gui username/password. Backend xac thuc bang BCrypt, tao session token, tra ve token.
2 localStorage React luu token vao localStorage duoi key "access_token".
3 Axios Interceptor Moi request HTTP, interceptor tu dong dinh kem header: Authorization: Bearer <token>
4 TokenAuthMiddleware Middleware doc header, tra cuu token trong DB, set ClaimsPrincipal voi Username va Role.
5 RequireAuthFilter ServiceFilter kiem tra User.Identity.IsAuthenticated. Tra 401 neu chua dang nhap.

6.3 Backend Middleware

// TokenAuthMiddleware.cs
public async Task InvokeAsync(HttpContext context, AppDbContext db)
{
    var authHeader = context.Request.Headers.Authorization.FirstOrDefault();

    if (!string.IsNullOrEmpty(authHeader)
        && authHeader.StartsWith("Bearer ", StringComparison.OrdinalIgnoreCase))
    {
        var token = authHeader["Bearer ".Length..].Trim();
        var session = await db.SessionTokens
            .Include(s => s.Account)
            .FirstOrDefaultAsync(s => s.Token == token);

        if (session is not null
            && session.ExpiresAt > DateTime.UtcNow
            && session.Account is not null)
        {
            var claims = new[] {
                new Claim(ClaimTypes.Name, session.Account.Username),
                new Claim(ClaimTypes.Role, session.Account.Role),
            };
            var identity = new ClaimsIdentity(claims, "SessionToken");
            context.User = new ClaimsPrincipal(identity);
        }
    }
    await _next(context);
}

6.4 Frontend Axios Interceptor

// http.ts
export const http = axios.create({
  baseURL: API_BASE,
  timeout: 20000,
  headers: { "ngrok-skip-browser-warning": "true" },
});

http.interceptors.request.use((config) => {
  const token = localStorage.getItem("access_token");
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;

});

7.1 Kien truc Translation Tables

Thay vi luu text truc tiep trong entity chinh, he thong su dung bang dich rieng biet (Translation Tables) voi quan he 1:N. Moi ban ghi co LanguageCode (vi, en, ja, ko, zh) va noi dung tuong ung. Cach tiep can nay cho phep mo rong ngon ngu khong gioi han ma khong can thay doi schema.

Entity gocTranslation TableUnique Constraint
Place PlaceTranslations [PlaceId, LanguageCode] - UNIQUE INDEX
Dish DishTranslations [DishId, LanguageCode] - UNIQUE INDEX 
// AppDbContext.cs - Fluent API
modelBuilder.Entity<PlaceTranslation>(entity => {
    entity.HasIndex(e => new { e.PlaceId, e.LanguageCode }).IsUnique();
    entity.Property(e => e.LanguageCode).HasMaxLength(10).IsRequired();
    entity.Property(e => e.Name).HasMaxLength(200).IsRequired();
    entity.Property(e => e.Description).HasMaxLength(2000);
});

7.2 Frontend - Query da ngon ngu

Frontend truyen tham so ?lang=vi hoac ?lang=en trong moi API call. Backend loc ban dich tuong ung va tra ve duy nhat ban dich phu hop. Khi nguoi dung chuyen ngon ngu tren UI, he thong goi lai API va cap nhat noi dung khong can reload trang. 

// tourService.ts
export const placesService = {
  getAll: (lang: string = 'en') =>
    http.get<PlaceDto[]>('/api/places', { params: { lang } })
        .then(r => r.data),
};

// TourMapPage.tsx - Fetch khi chuyen ngon ngu
useEffect(() => {
  placesService.getAll(lang).then(data => setPlaces(data));
}, [lang]);

7.3 Audio da ngon ngu

Khi nguoi dung chuyen ngon ngu, SPEECH_LANG_MAP tu dong chuyen doi locale cho Web Speech API. Dong thoi, tap narratedRef duoc reset de cho phep doc lai cac dia diem bang ngon ngu moi. 

// Reset khi chuyen ngon ngu
useEffect(() => {
  narratedRef.current.clear();      // Cho phep doc lai tat ca dia diem
  window.speechSynthesis.cancel();  // Dung giong doc hien tai
  setIsSpeaking(false);
}, [lang]);

Design Pattern & Key Constants

10.1 Backend Patterns

Repository Pattern

Tach biet logic truy van du lieu (Infrastructure) khoi logic nghiep vu (Application). Moi entity co cac interface rieng biet cho Command va Query (CQRS-lite): 

// Domain Layer - Interface
public interface IPlaceCommandRepository {
    Task<Place> CreateAsync(Place place);
    Task<Place> UpdateAsync(Place place);
    Task DeleteAsync(Guid id);
}

public interface IPlaceQueryRepository {
    Task<Place?> GetByIdAsync(Guid id, string languageCode);
    Task<List<Place>> GetAllAsync(string languageCode);
}

// Infrastructure Layer - Implementation
public class PlaceCommandRepository : IPlaceCommandRepository {
    private readonly AppDbContext _db;
    // ... EF Core implementation
}

Dependency Injection

// Program.cs - Service Registration
builder.Services.AddScoped<IPlaceCommandRepository, PlaceCommandRepository>();
builder.Services.AddScoped<IPlaceQueryRepository, PlaceQueryRepository>();
builder.Services.AddScoped<IDishCommandRepository, DishCommandRepository>();
builder.Services.AddScoped<IDishQueryRepository, DishQueryRepository>();
builder.Services.AddScoped<IOrderRepository, OrderRepository>();

builder.Services.AddScoped<PlaceCommandService>();
builder.Services.AddScoped<PlaceQueryService>();
builder.Services.AddScoped<DishCommandService>();
builder.Services.AddScoped<DishQueryService>();
builder.Services.AddScoped<OrderService>();

10.2 Frontend Patterns

Custom Hooks

Logic phuc tap duoc dong goi trong React Custom Hooks, tai su dung duoc va de test:

  • useGeolocation() - Theo doi vi tri GPS thoi gian thuc, tra ve position/error/isTracking.
  • useAuth() - Quan ly trang thai dang nhap, token, auto-validate khi reload trang.

Axios Interceptor Pattern

Mot diem duy nhat (http.ts) cau hinh baseURL, timeout, va tu dong dinh kem Bearer token + ngrok header cho moi request. Tat ca service file su dung chung instance nay.
